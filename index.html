def calculate_discount(price, discount_rate, user_type='regular'):
    """Calculate discount based on price, rate, and user type"""
    if price < 0:
        raise ValueError("Price cannot be negative")
    
    if not 0 <= discount_rate <= 1:
        raise ValueError("Discount rate must be between 0 and 1")
    
    base_discount = price * discount_rate
    
    # Premium users get additional 5% discount
    if user_type == 'premium':
        base_discount *= 1.05
    
    # Maximum discount cap
    max_discount = price * 0.5
    return min(base_discount, max_discount)

import hashlib
import secrets

def authenticate_user(username, password, stored_hash, salt):
    """Authenticate user with secure password hashing"""
    if not username or not password:
        return False
    
    # Hash the provided password with salt
    password_hash = hashlib.pbkdf2_hmac(
        'sha256',
        password.encode('utf-8'),
        salt.encode('utf-8'),
        100000
    )
    
    return secrets.compare_digest(password_hash, stored_hash)

üîß Troubleshooting

Common issues and solutions

üîë

API Key Issues

Problem: "Invalid API key" or authentication errors

Solution:

  • Verify your Gemini API key is correct
  • Check that the API key has proper permissions
  • Ensure the .env file is in the correct directory
  • Try regenerating the API key
‚ö†Ô∏è

Generation Failures

Problem: Tests not generating or poor quality output

Solution:

  • Check that source code is syntactically correct
  • Verify function signatures are well-formed
  • Ensure adequate function documentation
  • Try reducing coverage target temporarily
üêõ

Import Errors

Problem: Generated tests have import issues

Solution:

  • Check that source modules are in Python path
  • Verify relative import paths are correct
  • Ensure all dependencies are installed
  • Use absolute imports when possible
